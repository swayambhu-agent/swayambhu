# Mind Tab Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a "Mind" tab to the dashboard showing the cognitive system's state — samskaras, desires, experiences, operator health, and the flow between them.

**Architecture:** New `GET /mind` API endpoint batch-reads cognitive state from KV. SPA gets a new MindTab component with three sections: operator health bar, SVG flow graph, and expandable detail panels. Header bar gets cognitive health counters. All inline in the existing single-file SPA pattern.

**Tech Stack:** React 18 (browser Babel), Tailwind CSS, SVG for flow graph, Cloudflare Workers KV for data.

**Reference:** `docs/superpowers/specs/2026-04-01-mind-tab-design.md`

```javascript
if (path === "/mind") {
  const [
    samskaraKeys, desireKeys, experienceKeys,
    reflectSchedule, sessionCounter,
  ] = await Promise.all([
    kvListAll(env.KV, { prefix: "samskara:" }),
    kvListAll(env.KV, { prefix: "desire:" }),
    kvListAll(env.KV, { prefix: "experience:" }),
    env.KV.get("reflect:schedule:1", "json"),
    env.KV.get("session_counter", "json"),
  ]);

  // Batch read all values
  const allKeys = [
    ...samskaraKeys.map(k => k.name),
    ...desireKeys.map(k => k.name),
    ...experienceKeys.map(k => k.name).slice(-20), // last 20 experiences
  ];

  const values = {};
  await Promise.all(allKeys.map(async (key) => {
    const val = await env.KV.get(key, "json");
    if (val) values[key] = val;
  }));

  // Build response
  const samskaras = samskaraKeys
    .map(k => ({ key: k.name, ...values[k.name] }))
    .filter(s => s.pattern);

  const desires = desireKeys
    .map(k => ({ key: k.name, ...values[k.name] }))
    .filter(d => d.description || d.slug);

  const experiences = experienceKeys
    .slice(-20)
    .map(k => ({ key: k.name, ...values[k.name] }))
    .filter(e => e.narrative || e.action_taken)
    .reverse(); // newest first

  // Find latest deep-reflect output
  const reflectKeys = await kvListAll(env.KV, { prefix: "reflect:1:" });
  const latestReflectKey = reflectKeys
    .filter(k => !k.name.includes("schedule"))
    .sort((a, b) => b.name.localeCompare(a.name))[0];
  const latestReflect = latestReflectKey
    ? await env.KV.get(latestReflectKey.name, "json")
    : null;

  // Operator health
  const lastDrSession = reflectSchedule?.last_reflect_session || 0;
  const currentSession = sessionCounter || 0;
  const sessionsSinceDr = currentSession - lastDrSession;
  const nextDrDue = reflectSchedule?.after_sessions
    ? lastDrSession + reflectSchedule.after_sessions
    : null;

  const operatorHealth = {
    bootstrap_complete: samskaras.length > 0 || desires.length > 0,
    last_deep_reflect_session: lastDrSession,
    sessions_since_dr: sessionsSinceDr,
    next_dr_due: nextDrDue,
    last_reflect_output: latestReflect ? {
      session_id: latestReflect.session_id,
      reflection: latestReflect.reflection?.slice(0, 200),
      has_kv_operations: !!(latestReflect.kv_operations?.length),
    } : null,
  };

  return json({ samskaras, desires, experiences, operator_health: operatorHealth });
}
```

```javascript
function MindTab({ patronKey }) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  const [selected, setSelected] = useState(null); // {type: 'samskara'|'desire'|'experience', key: '...'}

  const loadData = useCallback(async () => {
    try {
      const result = await api('/mind', patronKey);
      setData(result);
      setError(null);
    } catch (e) {
      setError(e.message);
    }
    setLoading(false);
  }, [patronKey]);

  useEffect(() => { loadData(); }, [loadData]);

  if (loading) return <div className="p-8 text-gray-500">Loading cognitive state...</div>;
  if (error) return <div className="p-8 text-red-400">Error: {error}</div>;
  if (!data) return <div className="p-8 text-gray-500">No data</div>;

  return (
    <div className="flex flex-col h-full">
      <MindHealthBar health={data.operator_health} counts={{
        samskaras: data.samskaras.length,
        desires: data.desires.length,
        experiences: data.experiences.length,
      }} />
      <MindFlowGraph data={data} selected={selected} onSelect={setSelected} />
      <MindDetailPanels data={data} selected={selected} onSelect={setSelected} />
    </div>
  );
}
```

```javascript
function MindHealthBar({ health, counts }) {
  const bootstrap = health?.bootstrap_complete;
  const sincedr = health?.sessions_since_dr ?? '?';
  const nextdr = health?.next_dr_due;
  const currentSession = (health?.last_deep_reflect_session || 0) + sincedr;
  const sessionsUntilDr = nextdr ? Math.max(0, nextdr - currentSession) : '?';

  return (
    <div className="px-4 py-2 bg-bg-panel border-b border-border flex gap-5 flex-wrap items-center text-xs">
      <div className="flex items-center gap-1.5">
        <span style={{
          width: 7, height: 7, borderRadius: '50%', display: 'inline-block',
          background: bootstrap ? '#22c55e' : '#f59e0b',
        }} />
        <span className="text-gray-500">Bootstrap:</span>
        <span className="text-gray-200">{bootstrap ? 'Complete' : 'Pending'}</span>
      </div>

      <div className="text-gray-500">
        Last DR: <span className="text-deep font-semibold">{sincedr} sessions ago</span>
      </div>

      <div className="text-gray-500">
        Next DR: <span className="text-gray-200">~{sessionsUntilDr} sessions</span>
      </div>

      {health?.last_reflect_output && (
        <div className="text-gray-500">
          DR output: <span className="text-green-400">
            {health.last_reflect_output.has_kv_operations ? 'wrote KV' : 'observations only'}
          </span>
        </div>
      )}

      <div className="flex items-center gap-3 ml-auto text-gray-500">
        <span><span className="text-green-400">●</span> {counts.samskaras} samskaras</span>
        <span><span className="text-deep">●</span> {counts.desires} desires</span>
        <span><span className="text-cyan-400">●</span> {counts.experiences} experiences</span>
      </div>
    </div>
  );
}
```

- [ ] **Step 1: Replace the MindFlowGraph stub**

```javascript
function MindFlowGraph({ data, selected, onSelect }) {
  const { samskaras, desires, experiences } = data;

  // Layout constants
  const W = 900, H = 200;
  const cols = { exp: 80, sam: 340, des: 600, act: 810 };
  const nodeH = 28, nodeGap = 6, nodeW = { exp: 140, sam: 175, des: 165 };

  // Position nodes
  const expNodes = experiences.slice(0, 5).map((e, i) => ({
    ...e, x: 10, y: 32 + i * (nodeH + nodeGap), w: nodeW.exp,
  }));
  const samNodes = samskaras.map((s, i) => ({
    ...s, x: cols.sam - nodeW.sam / 2, y: 32 + i * (nodeH + nodeGap), w: nodeW.sam,
  }));
  const desNodes = desires.map((d, i) => ({
    ...d, x: cols.des - nodeW.des / 2, y: 32 + i * (nodeH + nodeGap + 6), w: nodeW.des,
  }));

  const isSelected = (type, key) => selected?.type === type && selected?.key === key;
  const dimmed = selected ? 0.15 : 1;

  // Color by strength
  const strengthColor = (s) => s > 0.7 ? '#22c55e' : s > 0.3 ? '#f59e0b' : '#ef4444';

  // Truncate text to fit node width
  const trunc = (t, max) => t && t.length > max ? t.slice(0, max - 1) + '…' : t;

  return (
    <div className="px-4 py-2 border-b border-border" style={{ background: 'rgba(255,255,255,0.01)' }}>
      <svg viewBox={`0 0 ${W} ${Math.max(H, 32 + Math.max(expNodes.length, samNodes.length, desNodes.length) * (nodeH + nodeGap) + 20)}`}
           style={{ width: '100%', height: 'auto', maxHeight: 220, cursor: 'default' }}>

        {/* Column labels */}
        <text x={cols.exp} y={14} fill="#666" fontSize="9" textAnchor="middle" fontFamily="monospace" letterSpacing="1">EXPERIENCES (ε)</text>
        <text x={cols.sam} y={14} fill="#666" fontSize="9" textAnchor="middle" fontFamily="monospace" letterSpacing="1">SAMSKARAS (s)</text>
        <text x={cols.des} y={14} fill="#666" fontSize="9" textAnchor="middle" fontFamily="monospace" letterSpacing="1">DESIRES (d)</text>

        {/* Connection lines: experiences → samskaras (session-based correlation) */}
        {expNodes.map((e, ei) =>
          samNodes.map((s, si) => (
            <line key={`es-${ei}-${si}`}
              x1={e.x + e.w} y1={e.y + nodeH/2}
              x2={s.x} y2={s.y + nodeH/2}
              stroke="#22c55e" strokeWidth="0.8"
              opacity={selected
                ? (isSelected('experience', e.key) || isSelected('samskara', s.key) ? 0.4 : 0.05)
                : 0.12}
            />
          ))
        )}

        {/* Connection lines: experiences → desires */}
        {expNodes.map((e, ei) =>
          desNodes.map((d, di) => (
            <line key={`ed-${ei}-${di}`}
              x1={e.x + e.w} y1={e.y + nodeH/2}
              x2={d.x} y2={d.y + nodeH/2}
              stroke="#a78bfa" strokeWidth="0.6"
              opacity={selected
                ? (isSelected('experience', e.key) || isSelected('desire', d.key) ? 0.3 : 0.04)
                : 0.08}
            />
          ))
        )}

        {/* Connection lines: desires → right edge (actions) */}
        {desNodes.map((d, di) => (
          <line key={`da-${di}`}
            x1={d.x + d.w} y1={d.y + nodeH/2}
            x2={W - 20} y2={d.y + nodeH/2}
            stroke="#a78bfa" strokeWidth="1"
            opacity={selected ? (isSelected('desire', d.key) ? 0.5 : 0.08) : 0.2}
            strokeDasharray="4,3"
          />
        ))}

        {/* Feedback loop arrow */}
        <path
          d={`M ${W-10} ${32 + nodeH/2} Q ${W} ${32+nodeH/2} ${W} ${H-8} L 10 ${H-8} Q 5 ${H-8} 5 ${H-20}`}
          stroke="#06b6d4" strokeWidth="1" opacity="0.15" fill="none" strokeDasharray="4,3"
        />
        <text x={W/2} y={H-2} fill="#06b6d4" fontSize="8" textAnchor="middle" opacity="0.3" fontFamily="monospace">→ new experiences</text>

        {/* Experience nodes */}
        {expNodes.map((e, i) => {
          const hi = isSelected('experience', e.key);
          return (
            <g key={e.key} onClick={() => onSelect(hi ? null : {type:'experience', key:e.key})} style={{cursor:'pointer'}}>
              <rect x={e.x} y={e.y} width={e.w} height={nodeH} rx={5}
                fill={hi ? 'rgba(6,182,212,0.25)' : 'rgba(6,182,212,0.1)'}
                stroke="#06b6d4" strokeWidth={hi ? 1.5 : 0.7} />
              <text x={e.x+4} y={e.y+11} fill="#06b6d4" fontSize="8" fontFamily="monospace">
                σ={e.surprise_score?.toFixed(1) ?? '?'}
              </text>
              <text x={e.x+4} y={e.y+22} fill="#888" fontSize="7.5">
                {trunc(e.narrative || e.action_taken || '(no narrative)', 22)}
              </text>
            </g>
          );
        })}

        {/* Samskara nodes */}
        {samNodes.map((s, i) => {
          const hi = isSelected('samskara', s.key);
          const col = strengthColor(s.strength);
          return (
            <g key={s.key} onClick={() => onSelect(hi ? null : {type:'samskara', key:s.key})} style={{cursor:'pointer'}}>
              <rect x={s.x} y={s.y} width={s.w} height={nodeH} rx={5}
                fill={hi ? `${col}22` : `${col}11`}
                stroke={col} strokeWidth={hi ? 1.5 : 0.7} />
              <text x={s.x+6} y={s.y+12} fill="#e5e5e5" fontSize="8.5">
                {trunc(s.pattern, 24)}
              </text>
              <text x={s.x + s.w - 6} y={s.y+22} fill={col} fontSize="8" textAnchor="end" fontWeight="bold">
                {s.strength?.toFixed(2)}
              </text>
              {/* Mini strength bar */}
              <rect x={s.x+6} y={s.y+18} width={60} height={3} rx={1} fill="rgba(255,255,255,0.08)" />
              <rect x={s.x+6} y={s.y+18} width={60 * (s.strength || 0)} height={3} rx={1} fill={col} />
            </g>
          );
        })}

        {/* Desire nodes */}
        {desNodes.map((d, i) => {
          const hi = isSelected('desire', d.key);
          const arrow = d.direction === 'avoidance' ? '↓' : '↑';
          const arrowCol = d.direction === 'avoidance' ? '#ef4444' : '#22c55e';
          return (
            <g key={d.key} onClick={() => onSelect(hi ? null : {type:'desire', key:d.key})} style={{cursor:'pointer'}}>
              <rect x={d.x} y={d.y} width={d.w} height={nodeH+4} rx={5}
                fill={hi ? 'rgba(167,139,250,0.2)' : 'rgba(167,139,250,0.08)'}
                stroke="#a78bfa" strokeWidth={hi ? 1.5 : 0.7} />
              <text x={d.x+6} y={d.y+13} fill={arrowCol} fontSize="12">{arrow}</text>
              <text x={d.x+20} y={d.y+13} fill="#e5e5e5" fontSize="8.5">
                {trunc(d.description || d.slug, 20)}
              </text>
              <text x={d.x+20} y={d.y+25} fill="#666" fontSize="7">
                {d.direction || 'approach'}
              </text>
            </g>
          );
        })}

        {/* S/D operator badge */}
        <rect x={cols.sam + 80} y={samNodes.length * (nodeH+nodeGap) / 2 + 25} width={28} height={14} rx={3}
          fill="rgba(167,139,250,0.2)" stroke="#a78bfa" strokeWidth="0.7" />
        <text x={cols.sam + 94} y={samNodes.length * (nodeH+nodeGap) / 2 + 35}
          fill="#a78bfa" fontSize="8" textAnchor="middle" fontWeight="bold">S D</text>

        {/* Empty state */}
        {samskaras.length === 0 && desires.length === 0 && experiences.length === 0 && (
          <text x={W/2} y={H/2} fill="#666" fontSize="12" textAnchor="middle">
            No cognitive state yet — waiting for first session
          </text>
        )}
      </svg>
    </div>
  );
}
```

```javascript
function MindDetailPanels({ data, selected, onSelect }) {
  if (!selected) {
    return (
      <div className="flex-1 flex items-center justify-center text-gray-600 text-xs p-8">
        Click a node in the graph to see details
      </div>
    );
  }

  const { type, key } = selected;

  if (type === 'samskara') {
    const s = data.samskaras.find(s => s.key === key);
    if (!s) return null;
    const col = s.strength > 0.7 ? '#22c55e' : s.strength > 0.3 ? '#f59e0b' : '#ef4444';
    return (
      <div className="flex-1 p-4 overflow-auto">
        <div style={{ padding: 16, background: `${col}08`, border: `1px solid ${col}25`, borderRadius: 8 }}>
          <div className="flex items-start justify-between mb-3">
            <div>
              <div className="text-xs text-gray-500 uppercase tracking-wider mb-1">Samskara</div>
              <div className="text-sm text-gray-200">{s.pattern}</div>
            </div>
            <div style={{ color: col, fontSize: 24, fontWeight: 'bold', fontFamily: 'monospace' }}>
              {s.strength?.toFixed(2)}
            </div>
          </div>
          <div className="flex gap-6 text-xs mt-3">
            <div>
              <span className="text-gray-500">Key: </span>
              <span className="text-gray-400 font-mono">{s.key}</span>
            </div>
          </div>
          <div className="mt-3 text-xs text-gray-500">
            Strength bar:
            <div style={{ height: 6, background: 'rgba(255,255,255,0.08)', borderRadius: 3, marginTop: 4 }}>
              <div style={{ width: `${(s.strength || 0) * 100}%`, height: '100%', background: col, borderRadius: 3 }} />
            </div>
          </div>
        </div>
      </div>
    );
  }

  if (type === 'desire') {
    const d = data.desires.find(d => d.key === key);
    if (!d) return null;
    const arrow = d.direction === 'avoidance' ? '↓' : '↑';
    const arrowCol = d.direction === 'avoidance' ? '#ef4444' : '#22c55e';
    return (
      <div className="flex-1 p-4 overflow-auto">
        <div style={{ padding: 16, background: 'rgba(167,139,250,0.06)', border: '1px solid rgba(167,139,250,0.15)', borderRadius: 8 }}>
          <div className="text-xs text-gray-500 uppercase tracking-wider mb-1">Desire</div>
          <div className="flex items-center gap-2 mb-3">
            <span style={{ color: arrowCol, fontSize: 18 }}>{arrow}</span>
            <span className="text-sm text-gray-200">{d.description || d.slug}</span>
          </div>
          <div className="flex gap-6 text-xs">
            <div>
              <span className="text-gray-500">Direction: </span>
              <span style={{ color: arrowCol }}>{d.direction || 'approach'}</span>
            </div>
            {d.source_principles && (
              <div>
                <span className="text-gray-500">Principles: </span>
                <span className="text-gray-400">{d.source_principles.join(', ')}</span>
              </div>
            )}
            {d.created_at && (
              <div>
                <span className="text-gray-500">Created: </span>
                <span className="text-gray-400">{new Date(d.created_at).toLocaleDateString()}</span>
              </div>
            )}
          </div>
          <div className="mt-2 text-xs">
            <span className="text-gray-500">Key: </span>
            <span className="text-gray-400 font-mono">{d.key}</span>
          </div>
        </div>
      </div>
    );
  }

  if (type === 'experience') {
    const e = data.experiences.find(e => e.key === key);
    if (!e) return null;
    return (
      <div className="flex-1 p-4 overflow-auto">
        <div style={{ padding: 16, background: 'rgba(6,182,212,0.06)', border: '1px solid rgba(6,182,212,0.15)', borderRadius: 8 }}>
          <div className="flex items-start justify-between mb-2">
            <div className="text-xs text-gray-500 uppercase tracking-wider">Experience</div>
            <div className="text-xs">
              <span className="text-cyan-400">σ={e.surprise_score?.toFixed(2) ?? '?'}</span>
              <span className="text-gray-600 mx-1">·</span>
              <span className="text-amber-400">salience={e.salience?.toFixed(2) ?? '?'}</span>
            </div>
          </div>
          {e.action_taken && (
            <div className="text-xs text-gray-400 mb-2">
              <span className="text-gray-500">Action: </span>{e.action_taken}
            </div>
          )}
          {e.narrative && (
            <div className="text-sm text-gray-300 leading-relaxed mb-3">{e.narrative}</div>
          )}
          {e.outcome && (
            <div className="text-xs text-gray-400">
              <span className="text-gray-500">Outcome: </span>{e.outcome}
            </div>
          )}
          <div className="mt-2 text-xs">
            <span className="text-gray-500">Key: </span>
            <span className="text-gray-400 font-mono">{e.key}</span>
            {e.timestamp && (
              <>
                <span className="text-gray-600 mx-2">·</span>
                <span className="text-gray-500">{new Date(e.timestamp).toLocaleString()}</span>
              </>
            )}
          </div>
        </div>
      </div>
    );
  }

  return null;
}
```
